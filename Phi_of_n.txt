<?php
$x = 100;
$i = 1;
$c = 0;
$factor=array();
$A=1;
echo "The factors of the " . $x . " are: ";
while ($i <= $x) {
    if ($x % $i == 0) {
        $factor[$c]=$i;
        echo $factor[$c] . " ";
        $flag=primeCheck($factor[$c]);
        if($flag==1){
        	$a=1-1/$factor[$c];
        	$A=$A*$a;
        }
        $c++;
    }
    ++$i;
}
echo "<br \>";
echo "Phi of ".$x." is: ".($A*$x);
function primeCheck($number){
    if ($number == 1)
    return 0;
    for ($i = 2; $i <= $number/2; $i++){
        if ($number % $i == 0)
            return 0;
    }
    return 1;
}
